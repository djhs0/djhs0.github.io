        # python的C3算法的实现

        在python3中，多继承的基石就是通过C3算法计算出的mro(方法解析顺序)来确定继承顺序。

        # c3算法的公式

        class A(object): pass

class B(object): pass

class C(object): pass

class E(A, B): pass

class F(B, C): pass

class G(E, F): pass
# 公式
mro(G) = [G] + merge(mro(E), mro(F), [E, F])
# 计算步骤
1. mro(E) = [E] + merge(mro(A), mro(B), [A, B])
- mro(A) = [A, O]
- mro(B) = [B, O]
- mro(E) = [E] + merge([A, O], [B, O], [A, B])
- mro(E) = [E, A] + merge([O], [B, O], [B])
- mro(E) = [E, A, B] + merge([O], [O], [])   # 虽然O在[O]中是第一个元素，但是在[B, O]不是第一个元素，所以不符合。
- mro(E) = [E, A, B, O]

2. mro(F) = [F] + merge(mro(B), mro(C), [B, C])
- mro(B) = [B, O]
- mro(C) = [C, O]
- mro(F) = [F] + merge([B, O], [C, O], [B, C])
- mro(F) = [F, B] + merge([O], [C, O], [C])
- mro(F) = [F, B, C] + merge([O], [O], [])
- mro(F) = [F, B, C, O]

mro(G) = [G] + merge([E, A, B, O], [F, B, C, O], [E, F])  
mro(G) = [G, E] + merge([A, B, O], [F, B, C, O], [F])  
mro(G) = [G, E, A] + merge([B, O], [F, B, C, O], [F])  
mro(G) = [G, E, A, F] + merge([B, O], [B, C, O], [])  
mro(G) = [G, E, A, F, B] + merge([O], [C, O], [])  
mro(G) = [G, E, A, F, B, C] + merge([O], [O], [])  
mro(G) = [G, E, A, F, B, C, O]  
最终class G的方法解析顺序为[G, E, A, F, B, C, O]

        # c3算法的代码实现

        def isok(lsts, target):
    """判断target是否有前驱"""
    flag = True
    for lst in lsts:
        if (target in lst) and (lst[0] != target):
            flag = False
            break
    return flag

def mymerge(lsts):
    res = []
    print(lsts)
    while [x for i in lsts for x in i]:  # 判断是否还有元素
        print(lsts)
        # 找出候选元素
        targets = []
        for i in lsts:
            if i:
                targets.append(i[0])
        is_DAG = False
        for tar in targets:
            if isok(lsts, tar):
                # 如果符合要求，就加入res列表中
                res.append(tar)
                # 然后从列表中删除此元素
                for i in lsts:
                    if tar in i:
                        i.remove(tar)
                is_DAG = True
                break
        if is_DAG is False:
            raise Exception("输入不合法")      
    return res

def mymro(cls):
    if cls is object:
        return [object]
    lsts = []
    for ob in cls.__bases__:
        lsts.append(mymro(ob))
    lsts.append(list(cls.__bases__))
    return [cls] + mymerge(lsts)

        # 理解

        在公式

        mro(G) = [G] + merge(mro(E), mro(F), [E, F])

        中，之所以在merge函数中最后要加入基类的列表[E, F]，是因为拓扑排序的结果是不固定的，但是加上[E, F]之后，拓扑排序的结果就被固定了。
