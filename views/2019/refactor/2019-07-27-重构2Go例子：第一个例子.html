      重构2Go例子：第一个例子

      djhs0 2019-7-27 重构 go

      # 原始代码

      在重构2的开篇，作者描述了一个例子

      戏剧演出团演出，客户指定演出的戏剧，剧团则根据戏剧类目和观看人数来向客户收付费

      原始代码中作者是使用javascript实现的，下面是使用go的实现

      package main

import (
	"errors"
	"fmt"
	"math"
)

// Performance 演出
type Performance struct {
	PlayID   string `json:"palyID"`
	Audience int32  `json:"audience"`
}

// Invoice 账单
type Invoice struct {
	Customer     string        `json:"customer"`
	Performances []Performance `json:"performances"`
}

// Play 戏曲字典
type Play map[string]string

// Plays 戏剧集合字典
type Plays map[string]Play

/*
 *原始函数：计算客户账单
 */
func statement(invoice Invoice, plays Plays) (string, error) {
	var totalAmount int32
	var volumeCredits int32
	result := fmt.Sprintf("Statement for %s\n", invoice.Customer)

	for _, perf := range invoice.Performances {
		play := plays[perf.PlayID]
		var thisAmount int32

		switch play["type"] {
		case "tragedy":
			thisAmount = 40000
			if perf.Audience > 30 {
				thisAmount += 1000 * (perf.Audience - 30)
			}
			break
		case "comedy":
			thisAmount = 30000
			if perf.Audience > 20 {
				thisAmount += 10000 + 500*(perf.Audience-20)
			}
			thisAmount += 300 * perf.Audience
			break
		default:
			return "", errors.New("没有这个类型的戏剧")
		}

		volumeCredits += int32(math.Max(float64(perf.Audience-30), 0))
		if play["type"] == "comedy" {
			volumeCredits += int32(math.Floor(float64(perf.Audience) / 5))
		}

		result += fmt.Sprintf("%s: $%d (%d seats)\n", play["name"], thisAmount/100, perf.Audience)
		totalAmount += thisAmount
	}
	result += fmt.Sprintf("Amount owed is $%d\n", totalAmount/100)
	result += fmt.Sprintf("You earned $%d credits\n", volumeCredits)
	return result, nil
}

      # 重构点

      # 将计算金额的功能提取成函数

      在原始函数中，计算金额的switch代码块有明显的功能属性，同时也可以清楚的提取入参play，perf，出参thisAmount，所以将switch代码块提取成函数。

      在这个函数中，突然明白go错误处理的局限性，由于计算金额的函数中有可能出现例外情况，所以需要处理异常，导致amountFor函数需要返回两个值，这样它的返回值必须有临时变量接收。

      // 提炼switch为函数
func amountFor(perf Performance, play Play) (int32, error) {
	var result int32
	switch play["type"] {
	case "tragedy":
		result = 40000
		if perf.Audience > 30 {
			result += 1000 * (perf.Audience - 30)
		}
		break
	case "comedy":
		result = 30000
		if perf.Audience > 20 {
			result += 10000 + 500*(perf.Audience-20)
		}
		result += 300 * perf.Audience
		break
	default:
		return -1, errors.New("没有这个类型的戏剧")
	}

	return result, nil
}

      # 将计算观众量积分功能提取成函数

      在原始的代码中，计算观众量积分的代码块功能明确，有着明确的入参perf，play，出参volumeCredits，所以将其提取成函数

      // 计算观众量积分
func volumeCreditsFor(perf Performance, play Play) int32 {
	var result int32
	result += int32(math.Max(float64(perf.Audience-30), 0))
	if play["type"] == "comedy" {
		result += int32(math.Floor(float64(perf.Audience) / 5))
	}
	return result
}

      # 拆分循环，移除观众量积分总和

      在原始代码中，计算金额和计算观众量积分两个功能是在一个循环中同时完成的，虽然这样可以避免一次循环，但是将他们拆分出来是不是更清晰呢？在书中作者认为这种重构手法是值得的，而且清晰的代码更有利于性能调优。

      func totalVolumeCredits(invoice Invoice, plays Plays) int32 {
	var result int32
	for _, perf := range invoice.Performances {
		play := plays[perf.PlayID]
		result += volumeCreditsFor(perf, play)
	}
	return result
}

      # 最终成果

      在看书的过程中，自己能体会的就是上面三个重构点，其他的比较高深的重构技巧现在还不能体会。 最大的体会是发现go的错误处理的不便之处，如果有try-catch的话就不需要返回err了，直接在外层catch就好了

      func statement2(invoice Invoice, plays Plays) (string, error) {
	var totalAmount int32
	result := fmt.Sprintf("Statement for %s\n", invoice.Customer)

	for _, perf := range invoice.Performances {
		play := plays[perf.PlayID]
		thisAmount, err := amountFor(perf, play)
		if err != nil {
			return "", err
		}

		result += fmt.Sprintf("%s: $%d (%d seats)\n", play["name"], thisAmount/100, perf.Audience)
		totalAmount += thisAmount
	}
	// 由于go的错误处理方式，amountFor有其他情况，
	// 必须处理错误不能只返回一个值，所以不能提炼计算totalAmount功能为函数

	result += fmt.Sprintf("Amount owed is $%d\n", totalAmount/100)
	result += fmt.Sprintf("You earned $%d credits\n", totalVolumeCredits(invoice, plays))
	return result, nil
}
